83      void register(NioSocketChannel channel, ChannelFuture future) {
84  
85          boolean server = !(channel instanceof NioClientSocketChannel);
86          Runnable registerTask = new RegisterTask(channel, future, server);
87          Selector selector;
88  
89          synchronized (startStopLock) {
90              if (!started) {
91                  // Open a selector if this worker didn't start yet.
92                  try {
93                      this.selector = selector = Selector.open();
94                  } catch (Throwable t) {
95                      throw new ChannelException(
96                              "Failed to create a selector.", t);
97                  }
98  
99                  // Start the worker thread with the new Selector.
100                 String threadName =
101                     (server ? "New I/O server worker #"
102                             : "New I/O client worker #") + bossId + '-' + id;
103 
104                 boolean success = false;
105                 try {
106                     executor.execute(
107                             new IoWorkerRunnable(
108                                     new ThreadRenamingRunnable(this, threadName)));
109                     success = true;
110                 } finally {
111                     if (!success) {
112                         // Release the Selector if the execution fails.
113                         try {
114                             selector.close();
115                         } catch (Throwable t) {
116                             logger.warn("Failed to close a selector.", t);
117                         }
118                         this.selector = selector = null;
119                         // The method will return to the caller at this point.
120                     }
121                 }
122             } else {
123                 // Use the existing selector if this worker has been started.
124                 selector = this.selector;
125             }
126 
127             assert selector != null && selector.isOpen();
128 
129             started = true;
130             boolean offered = registerTaskQueue.offer(registerTask);
131             assert offered;
132         }
133 
134         if (wakenUp.compareAndSet(false, true)) {
135             selector.wakeup();
136         }
137     }

139     public void run() {
140         thread = Thread.currentThread();
141 
142         boolean shutdown = false;
143         Selector selector = this.selector;
144         for (;;) {
145             wakenUp.set(false);
146 
147             if (CONSTRAINT_LEVEL != 0) {
148                 selectorGuard.writeLock().lock();
149                     // This empty synchronization block prevents the selector
150                     // from acquiring its lock.
151                 selectorGuard.writeLock().unlock();
152             }
153 
154             try {
155                 SelectorUtil.select(selector);
156 
157                 // 'wakenUp.compareAndSet(false, true)' is always evaluated
158                 // before calling 'selector.wakeup()' to reduce the wake-up
159                 // overhead. (Selector.wakeup() is an expensive operation.)
160                 //
161                 // However, there is a race condition in this approach.
162                 // The race condition is triggered when 'wakenUp' is set to
163                 // true too early.
164                 //
165                 // 'wakenUp' is set to true too early if:
166                 // 1) Selector is waken up between 'wakenUp.set(false)' and
167                 //    'selector.select(...)'. (BAD)
168                 // 2) Selector is waken up between 'selector.select(...)' and
169                 //    'if (wakenUp.get()) { ... }'. (OK)
170                 //
171                 // In the first case, 'wakenUp' is set to true and the
172                 // following 'selector.select(...)' will wake up immediately.
173                 // Until 'wakenUp' is set to false again in the next round,
174                 // 'wakenUp.compareAndSet(false, true)' will fail, and therefore
175                 // any attempt to wake up the Selector will fail, too, causing
176                 // the following 'selector.select(...)' call to block
177                 // unnecessarily.
178                 //
179                 // To fix this problem, we wake up the selector again if wakenUp
180                 // is true immediately after selector.select(...).
181                 // It is inefficient in that it wakes up the selector for both
182                 // the first case (BAD - wake-up required) and the second case
183                 // (OK - no wake-up required).
184 
185                 if (wakenUp.get()) {
186                     selector.wakeup();
187                 }
188 
189                 cancelledKeys = 0;
190                 processRegisterTaskQueue();
191                 processWriteTaskQueue();
192                 processSelectedKeys(selector.selectedKeys());
193 
194                 // Exit the loop when there's nothing to handle.
195                 // The shutdown flag is used to delay the shutdown of this
196                 // loop to avoid excessive Selector creation when
197                 // connections are registered in a one-by-one manner instead of
198                 // concurrent manner.
199                 if (selector.keys().isEmpty()) {
200                     if (shutdown ||
201                         executor instanceof ExecutorService && ((ExecutorService) executor).isShutdown()) {
202 
203                         synchronized (startStopLock) {
204                             if (registerTaskQueue.isEmpty() && selector.keys().isEmpty()) {
205                                 started = false;
206                                 try {
207                                     selector.close();
208                                 } catch (IOException e) {
209                                     logger.warn(
210                                             "Failed to close a selector.", e);
211                                 } finally {
212                                     this.selector = null;
213                                 }
214                                 break;
215                             } else {
216                                 shutdown = false;
217                             }
218                         }
219                     } else {
220                         // Give one more second.
221                         shutdown = true;
222                     }
223                 } else {
224                     shutdown = false;
225                 }
226             } catch (Throwable t) {
227                 logger.warn(
228                         "Unexpected exception in the selector loop.", t);
229 
230                 // Prevent possible consecutive immediate failures that lead to
231                 // excessive CPU consumption.
232                 try {
233                     Thread.sleep(1000);
234                 } catch (InterruptedException e) {
235                     // Ignore.
236                 }
237             }
238         }
239     }

242         for (;;) {
243             final Runnable task = registerTaskQueue.poll();
244             if (task == null) {

249             cleanUpCancelledKeys();

265     private void processSelectedKeys(Set<SelectionKey> selectedKeys) throws IOException {
266         for (Iterator<SelectionKey> i = selectedKeys.iterator(); i.hasNext();) {
267             SelectionKey k = i.next();
268             i.remove();
269             try {
270                 int readyOps = k.readyOps();
271                 if ((readyOps & SelectionKey.OP_READ) != 0 || readyOps == 0) {
272                     if (!read(k)) {
273                         // Connection already closed - no need to handle write.
274                         continue;
275                     }
276                 }
277                 if ((readyOps & SelectionKey.OP_WRITE) != 0) {
278                     writeFromSelectorLoop(k);
279                 }
280             } catch (CancelledKeyException e) {
281                 close(k);
282             }
283 
284             if (cleanUpCancelledKeys()) {
285                 break; // break the loop to avoid ConcurrentModificationException
286             }
287         }
288     }

290     private boolean cleanUpCancelledKeys() throws IOException {
291         if (cancelledKeys >= CLEANUP_INTERVAL) {
292             cancelledKeys = 0;
293             selector.selectNow();
294             return true;
295         }
296         return false;
297     }

299     private boolean read(SelectionKey k) {
300         final SocketChannel ch = (SocketChannel) k.channel();
301         final NioSocketChannel channel = (NioSocketChannel) k.attachment();
302 
303         final ReceiveBufferSizePredictor predictor =
304             channel.getConfig().getReceiveBufferSizePredictor();
305         final int predictedRecvBufSize = predictor.nextReceiveBufferSize();
306 
307         int ret = 0;
308         int readBytes = 0;
309         boolean failure = true;
310 
311         ByteBuffer bb = recvBufferPool.acquire(predictedRecvBufSize);
312         try {
313             while ((ret = ch.read(bb)) > 0) {
314                 readBytes += ret;
315                 if (!bb.hasRemaining()) {
316                     break;
317                 }
318             }
319             failure = false;
320         } catch (ClosedChannelException e) {
321             // Can happen, and does not need a user attention.
322         } catch (Throwable t) {
323             fireExceptionCaught(channel, t);
324         }
325 
326         if (readBytes > 0) {
327             bb.flip();
328 
329             final ChannelBufferFactory bufferFactory =
330                 channel.getConfig().getBufferFactory();
331             final ChannelBuffer buffer = bufferFactory.getBuffer(
332                     bb.order(bufferFactory.getDefaultOrder()));
333 
334             recvBufferPool.release(bb);
335 
336             // Update the predictor.
337             predictor.previousReceiveBufferSize(readBytes);
338 
339             // Fire the event.
340             fireMessageReceived(channel, buffer);
341         } else {
342             recvBufferPool.release(bb);
343         }
344 
345         if (ret < 0 || failure) {
346             close(channel, succeededFuture(channel));
347             return false;
348         }
349 
350         return true;
351     }

358     void writeFromUserCode(final NioSocketChannel channel) {
359         if (!channel.isConnected()) {
360             cleanUpWriteBuffer(channel);
361             return;
362         }
363 
364         if (scheduleWriteIfNecessary(channel)) {
365             return;
366         }
367 
368         // From here, we are sure Thread.currentThread() == workerThread.
369 
370         if (channel.writeSuspended) {
371             return;
372         }
373 
374         if (channel.inWriteNowLoop) {
375             return;
376         }
377 
378         write0(channel);
379     }

393     private boolean scheduleWriteIfNecessary(final NioSocketChannel channel) {
394         final Thread currentThread = Thread.currentThread();
395         final Thread workerThread = thread;
396         if (currentThread != workerThread) {
397             if (channel.writeTaskInTaskQueue.compareAndSet(false, true)) {
398                 boolean offered = writeTaskQueue.offer(channel.writeTask);
399                 assert offered;
400             }
401 
402             if (!(channel instanceof NioAcceptedSocketChannel) ||
403                 ((NioAcceptedSocketChannel) channel).bossThread != currentThread) {
404                 final Selector workerSelector = selector;
405                 if (workerSelector != null) {
406                     if (wakenUp.compareAndSet(false, true)) {
407                         workerSelector.wakeup();
408                     }
409                 }
410             } else {
411                 // A write request can be made from an acceptor thread (boss)
412                 // when a user attempted to write something in:
413                 //
414                 //   * channelOpen()
415                 //   * channelBound()
416                 //   * channelConnected().
417                 //
418                 // In this case, there's no need to wake up the selector because
419                 // the channel is not even registered yet at this moment.
420             }
421 
422             return true;
423         }
424 
425         return false;
426     }

428     private void write0(NioSocketChannel channel) {
429         boolean open = true;
430         boolean addOpWrite = false;
431         boolean removeOpWrite = false;
432 
433         long writtenBytes = 0;
434 
435         final SocketSendBufferPool sendBufferPool = this.sendBufferPool;
436         final SocketChannel ch = channel.socket;
437         final Queue<MessageEvent> writeBuffer = channel.writeBuffer;
438         final int writeSpinCount = channel.getConfig().getWriteSpinCount();
439         synchronized (channel.writeLock) {
440             channel.inWriteNowLoop = true;
441             for (;;) {
442                 MessageEvent evt = channel.currentWriteEvent;
443                 SendBuffer buf;
444                 if (evt == null) {
445                     if ((channel.currentWriteEvent = evt = writeBuffer.poll()) == null) {
446                         removeOpWrite = true;
447                         channel.writeSuspended = false;
448                         break;
449                     }
450 
451                     channel.currentWriteBuffer = buf = sendBufferPool.acquire(evt.getMessage());
452                 } else {
453                     buf = channel.currentWriteBuffer;
454                 }
455 
456                 ChannelFuture future = evt.getFuture();
457                 try {
458                     long localWrittenBytes = 0;
459                     for (int i = writeSpinCount; i > 0; i --) {
460                         localWrittenBytes = buf.transferTo(ch);
461                         if (localWrittenBytes != 0) {
462                             writtenBytes += localWrittenBytes;
463                             break;
464                         }
465                     }
466 
467                     if (buf.finished()) {
468                         // Successful write - proceed to the next message.
469                         buf.release();
470                         channel.currentWriteEvent = null;
471                         channel.currentWriteBuffer = null;
472                         evt = null;
473                         buf = null;
474                         future.setSuccess();
475                     } else {
476                         // Not written fully - perhaps the kernel buffer is full.
477                         addOpWrite = true;
478                         channel.writeSuspended = true;
479 
480                         if (localWrittenBytes > 0) {
481                             // Notify progress listeners if necessary.
482                             future.setProgress(
483                                     localWrittenBytes,
484                                     buf.writtenBytes(), buf.totalBytes());
485                         }
486                         break;
487                     }
488                 } catch (AsynchronousCloseException e) {
489                     // Doesn't need a user attention - ignore.
490                 } catch (Throwable t) {
491                     buf.release();
492                     channel.currentWriteEvent = null;
493                     channel.currentWriteBuffer = null;
494                     buf = null;
495                     evt = null;
496                     future.setFailure(t);
497                     fireExceptionCaught(channel, t);
498                     if (t instanceof IOException) {
499                         open = false;
500                         close(channel, succeededFuture(channel));
501                     }
502                 }
503             }
504             channel.inWriteNowLoop = false;
505         }
506 
507         fireWriteComplete(channel, writtenBytes);
508 
509         if (open) {
510             if (addOpWrite) {
511                 setOpWrite(channel);
512             } else if (removeOpWrite) {
513                 clearOpWrite(channel);
514             }
515         }
516     }

518     private void setOpWrite(NioSocketChannel channel) {
519         Selector selector = this.selector;
520         SelectionKey key = channel.socket.keyFor(selector);
521         if (key == null) {
522             return;
523         }
524         if (!key.isValid()) {
525             close(key);
526             return;
527         }
528         int interestOps;
529         boolean changed = false;
530 
531         // interestOps can change at any time and at any thread.
532         // Acquire a lock to avoid possible race condition.
533         synchronized (channel.interestOpsLock) {
534             interestOps = channel.getRawInterestOps();
535             if ((interestOps & SelectionKey.OP_WRITE) == 0) {
536                 interestOps |= SelectionKey.OP_WRITE;
537                 key.interestOps(interestOps);
538                 changed = true;
539             }
540         }
541 
542         if (changed) {
543             channel.setRawInterestOpsNow(interestOps);
544         }
545     }

576     void close(NioSocketChannel channel, ChannelFuture future) {
577         boolean connected = channel.isConnected();
578         boolean bound = channel.isBound();
579         try {
580             channel.socket.close();
581             cancelledKeys ++;
582 
583             if (channel.setClosed()) {
584                 future.setSuccess();
585                 if (connected) {
586                     fireChannelDisconnected(channel);
587                 }
588                 if (bound) {
589                     fireChannelUnbound(channel);
590                 }
591 
592                 cleanUpWriteBuffer(channel);
593                 fireChannelClosed(channel);
594             } else {
595                 future.setSuccess();
596             }
597         } catch (Throwable t) {
598             future.setFailure(t);
599             fireExceptionCaught(channel, t);
600         }
601     }

603     private void cleanUpWriteBuffer(NioSocketChannel channel) {
604         Exception cause = null;
605         boolean fireExceptionCaught = false;
606 
607         // Clean up the stale messages in the write buffer.
608         synchronized (channel.writeLock) {
609             MessageEvent evt = channel.currentWriteEvent;
610             if (evt != null) {
611                 // Create the exception only once to avoid the excessive overhead
612                 // caused by fillStackTrace.
613                 if (channel.isOpen()) {
614                     cause = new NotYetConnectedException();
615                 } else {
616                     cause = new ClosedChannelException();
617                 }
618 
619                 ChannelFuture future = evt.getFuture();
620                 channel.currentWriteBuffer.release();
621                 channel.currentWriteBuffer = null;
622                 channel.currentWriteEvent = null;
623                 evt = null;
624                 future.setFailure(cause);
625                 fireExceptionCaught = true;
626             }
627 
628             Queue<MessageEvent> writeBuffer = channel.writeBuffer;
629             if (!writeBuffer.isEmpty()) {
630                 // Create the exception only once to avoid the excessive overhead
631                 // caused by fillStackTrace.
632                 if (cause == null) {
633                     if (channel.isOpen()) {
634                         cause = new NotYetConnectedException();
635                     } else {
636                         cause = new ClosedChannelException();
637                     }
638                 }
639 
640                 for (;;) {
641                     evt = writeBuffer.poll();
642                     if (evt == null) {
643                         break;
644                     }
645                     evt.getFuture().setFailure(cause);
646                     fireExceptionCaught = true;
647                 }
648             }
649         }
650 
651         if (fireExceptionCaught) {
652             fireExceptionCaught(channel, cause);
653         }
654     }

656     void setInterestOps(
657             NioSocketChannel channel, ChannelFuture future, int interestOps) {
658         boolean changed = false;
659         try {
660             // interestOps can change at any time and at any thread.
661             // Acquire a lock to avoid possible race condition.
662             synchronized (channel.interestOpsLock) {
663                 Selector selector = this.selector;
664                 SelectionKey key = channel.socket.keyFor(selector);
665 
666                 if (key == null || selector == null) {
667                     // Not registered to the worker yet.
668                     // Set the rawInterestOps immediately; RegisterTask will pick it up.
669                     channel.setRawInterestOpsNow(interestOps);
670                     return;
671                 }
672 
673                 // Override OP_WRITE flag - a user cannot change this flag.
674                 interestOps &= ~Channel.OP_WRITE;
675                 interestOps |= channel.getRawInterestOps() & Channel.OP_WRITE;
676 
677                 switch (CONSTRAINT_LEVEL) {
678                 case 0:
679                     if (channel.getRawInterestOps() != interestOps) {
680                         key.interestOps(interestOps);
681                         if (Thread.currentThread() != thread &&
682                             wakenUp.compareAndSet(false, true)) {
683                             selector.wakeup();
684                         }
685                         changed = true;
686                     }
687                     break;
688                 case 1:
689                 case 2:
690                     if (channel.getRawInterestOps() != interestOps) {
691                         if (Thread.currentThread() == thread) {
692                             key.interestOps(interestOps);
693                             changed = true;
694                         } else {
695                             selectorGuard.readLock().lock();
696                             try {
697                                 if (wakenUp.compareAndSet(false, true)) {
698                                     selector.wakeup();
699                                 }
700                                 key.interestOps(interestOps);
701                                 changed = true;
702                             } finally {
703                                 selectorGuard.readLock().unlock();
704                             }
705                         }
706                     }
707                     break;
708                 default:
709                     throw new Error();
710                 }
711             }
712 
713             future.setSuccess();
714             if (changed) {
715                 channel.setRawInterestOpsNow(interestOps);
716                 fireChannelInterestChanged(channel);
717             }
718         } catch (CancelledKeyException e) {
719             // setInterestOps() was called on a closed channel.
720             ClosedChannelException cce = new ClosedChannelException();
721             future.setFailure(cce);
722             fireExceptionCaught(channel, cce);
723         } catch (Throwable t) {
724             future.setFailure(t);
725             fireExceptionCaught(channel, t);
726         }
727     }

729     private final class RegisterTask implements Runnable {
730         private final NioSocketChannel channel;
731         private final ChannelFuture future;
732         private final boolean server;
733 
734         RegisterTask(
735                 NioSocketChannel channel, ChannelFuture future, boolean server) {
736 
737             this.channel = channel;
738             this.future = future;
739             this.server = server;
740         }
741 
742         public void run() {
743             SocketAddress localAddress = channel.getLocalAddress();
744             SocketAddress remoteAddress = channel.getRemoteAddress();
745             if (localAddress == null || remoteAddress == null) {
746                 if (future != null) {
747                     future.setFailure(new ClosedChannelException());
748                 }
749                 close(channel, succeededFuture(channel));
750                 return;
751             }
752 
753             try {
754                 if (server) {
755                     channel.socket.configureBlocking(false);
756                 }
757 
758                 synchronized (channel.interestOpsLock) {
759                     channel.socket.register(
760                             selector, channel.getRawInterestOps(), channel);
761                 }
762                 if (future != null) {
763                     channel.setConnected();
764                     future.setSuccess();
765                 }
766             } catch (IOException e) {
767                 if (future != null) {
768                     future.setFailure(e);
769                 }
770                 close(channel, succeededFuture(channel));
771                 if (!(e instanceof ClosedChannelException)) {
772                     throw new ChannelException(
773                             "Failed to register a socket to the selector.", e);
774                 }
775             }
776 
777             if (!server) {
778                 if (!((NioClientSocketChannel) channel).boundManually) {
779                     fireChannelBound(channel, localAddress);
780                 }
781                 fireChannelConnected(channel, remoteAddress);
782             }
783         }
