44  /**
45   * A {@link ThreadPoolExecutor} which blocks the task submission when there's
46   * too many tasks in the queue.
47   * <p>
48   * Both per-{@link Channel} and per-{@link Executor} limitation can be applied.
49   * If the total size of the unprocessed tasks (i.e. {@link Runnable}s) exceeds
50   * either per-{@link Channel} or per-{@link Executor} threshold, any further
51   * {@link #execute(Runnable)} call will block until the tasks in the queue
52   * are processed so that the total size goes under the threshold.
53   * <p>
54   * {@link ObjectSizeEstimator} is used to calculate the size of each task.
55   * <p>
56   * Please note that this executor does not maintain the order of the
57   * {@link ChannelEvent}s for the same {@link Channel}.  For example,
58   * you can even receive a {@code "channelClosed"} event before a
59   * {@code "messageReceived"} event, as depicted by the following diagram.
60   *
61   * For example, the events can be processed as depicted below:
62   *
63   * <pre>
64   *           --------------------------------&gt; Timeline --------------------------------&gt;
65   *
66   * Thread X: --- Channel A (Event 2) --- Channel A (Event 1) ---------------------------&gt;
67   *
68   * Thread Y: --- Channel A (Event 3) --- Channel B (Event 2) --- Channel B (Event 3) ---&gt;
69   *
70   * Thread Z: --- Channel B (Event 1) --- Channel B (Event 4) --- Channel A (Event 4) ---&gt;
71   * </pre>
72   *
73   * To maintain the event order, you must use {@link OrderedMemoryAwareThreadPoolExecutor}.
74   *
75   * @author <a href="http://www.jboss.org/netty/">The Netty Project</a>
76   * @author <a href="http://gleamynode.net/">Trustin Lee</a>
77   *
78   * @version $Rev$, $Date$
79   *
80   * @apiviz.has org.jboss.netty.util.ObjectSizeEstimator oneway - -
81   * @apiviz.has org.jboss.netty.handler.execution.ChannelEventRunnable oneway - - executes
82   */
83  public class MemoryAwareThreadPoolExecutor extends ThreadPoolExecutor {

165             int corePoolSize, long maxChannelMemorySize, long maxTotalMemorySize,
166             long keepAliveTime, TimeUnit unit, ObjectSizeEstimator objectSizeEstimator,
167             ThreadFactory threadFactory) {
168 
169         super(corePoolSize, corePoolSize, keepAliveTime, unit,
170               new LinkedTransferQueue<Runnable>(), threadFactory, new NewThreadRunsPolicy());
171 
172         if (objectSizeEstimator == null) {
173             throw new NullPointerException("objectSizeEstimator");
174         }
175         if (maxChannelMemorySize < 0) {
176             throw new IllegalArgumentException(
177                     "maxChannelMemorySize: " + maxChannelMemorySize);
178         }
179         if (maxTotalMemorySize < 0) {
180             throw new IllegalArgumentException(
181                     "maxTotalMemorySize: " + maxTotalMemorySize);
182         }
183 
184         // Call allowCoreThreadTimeOut(true) using reflection
185         // because it is not supported in Java 5.
186         try {
187             Method m = getClass().getMethod("allowCoreThreadTimeOut", new Class[] { boolean.class });
188             m.invoke(this, Boolean.TRUE);
189         } catch (Throwable t) {
190             // Java 5
191             logger.debug(
192                     "ThreadPoolExecutor.allowCoreThreadTimeOut() is not " +
193                     "supported in this platform.");
194         }
195 
196         settings = new Settings(
197                 objectSizeEstimator, maxChannelMemorySize, maxTotalMemorySize);
198 
199         // Misuse check
200         misuseDetector.increase();
201     }

283     @Override
284     public void execute(Runnable command) {
285         if (!(command instanceof ChannelEventRunnable)) {
286             command = new MemoryAwareRunnable(command);
287         }
288 
289         boolean pause = increaseCounter(command);
290         doExecute(command);
291         if (pause) {
292             //System.out.println("ACQUIRE: " + command);
293             semaphore.acquireUninterruptibly();
294         }
295     }

327     protected boolean increaseCounter(Runnable task) {
328         if (!shouldCount(task)) {
329             return false;
330         }
331 
332         Settings settings = this.settings;
333         long maxTotalMemorySize = settings.maxTotalMemorySize;
334         long maxChannelMemorySize = settings.maxChannelMemorySize;
335 
336         int increment = settings.objectSizeEstimator.estimateSize(task);
337         long totalCounter = this.totalCounter.addAndGet(increment);
338 
339         if (task instanceof ChannelEventRunnable) {
340             ChannelEventRunnable eventTask = (ChannelEventRunnable) task;
341             eventTask.estimatedSize = increment;
342             Channel channel = eventTask.getEvent().getChannel();
343             long channelCounter = getChannelCounter(channel).addAndGet(increment);
344             //System.out.println("IC: " + channelCounter + ", " + increment);
345             if (maxChannelMemorySize != 0 && channelCounter >= maxChannelMemorySize && channel.isOpen()) {
346                 if (channel.isReadable()) {
347                     //System.out.println("UNREADABLE");
348                     ChannelHandlerContext ctx = eventTask.getContext();
349                     if (ctx.getHandler() instanceof ExecutionHandler) {
350                         // readSuspended = true;
351                         ctx.setAttachment(Boolean.TRUE);
352                     }
353                     channel.setReadable(false);
354                 }
355             }
356         } else {
357             ((MemoryAwareRunnable) task).estimatedSize = increment;
358         }
359 
360         //System.out.println("I: " + totalCounter + ", " + increment);
361         return maxTotalMemorySize != 0 && totalCounter >= maxTotalMemorySize;
362     }
363 
364     protected void decreaseCounter(Runnable task) {
365         if (!shouldCount(task)) {
366             return;
367         }
368 
369         Settings settings = this.settings;
370         long maxTotalMemorySize = settings.maxTotalMemorySize;
371         long maxChannelMemorySize = settings.maxChannelMemorySize;
372 
373         int increment;
374         if (task instanceof ChannelEventRunnable) {
375             increment = ((ChannelEventRunnable) task).estimatedSize;
376         } else {
377             increment = ((MemoryAwareRunnable) task).estimatedSize;
378         }
379 
380         long totalCounter = this.totalCounter.addAndGet(-increment);
381 
382         //System.out.println("D: " + totalCounter + ", " + increment);
383         if (maxTotalMemorySize != 0 && totalCounter + increment >= maxTotalMemorySize) {
384             //System.out.println("RELEASE: " + task);
385             semaphore.release();
386         }
387 
388         if (task instanceof ChannelEventRunnable) {
389             ChannelEventRunnable eventTask = (ChannelEventRunnable) task;
390             Channel channel = eventTask.getEvent().getChannel();
391             long channelCounter = getChannelCounter(channel).addAndGet(-increment);
392             //System.out.println("DC: " + channelCounter + ", " + increment);
393             if (maxChannelMemorySize != 0 && channelCounter < maxChannelMemorySize && channel.isOpen()) {
394                 if (!channel.isReadable()) {
395                     //System.out.println("READABLE");
396                     ChannelHandlerContext ctx = eventTask.getContext();
397                     if (ctx.getHandler() instanceof ExecutionHandler) {
398                         // readSuspended = false;
399                         ctx.setAttachment(null);
400                     }
401                     channel.setReadable(true);
402                 }
403             }
404         }
405     }

407     private AtomicLong getChannelCounter(Channel channel) {
408         AtomicLong counter = channelCounters.get(channel);
409         if (counter == null) {
410             counter = new AtomicLong();
411             AtomicLong oldCounter = channelCounters.putIfAbsent(channel, counter);
412             if (oldCounter != null) {
413                 counter = oldCounter;
414             }
415         }
416 
417         // Remove the entry when the channel closes.
418         if (!channel.isOpen()) {
419             channelCounters.remove(channel);
420         }
421         return counter;
422     }

424     /**
425      * Returns {@code true} if and only if the specified {@code task} should
426      * be counted to limit the global and per-channel memory consumption.
427      * To override this method, you must call {@code super.shouldCount()} to
428      * make sure important tasks are not counted.
429      */
430     protected boolean shouldCount(Runnable task) {
431         if (task instanceof ChannelEventRunnable) {
432             ChannelEventRunnable r = (ChannelEventRunnable) task;
433             ChannelEvent e = r.getEvent();
434             if (e instanceof WriteCompletionEvent) {
435                 return false;
436             } else if (e instanceof ChannelStateEvent) {
437                 if (((ChannelStateEvent) e).getState() == ChannelState.INTEREST_OPS) {
438                     return false;
439                 }
440             }
441         }
442         return true;
443     }
