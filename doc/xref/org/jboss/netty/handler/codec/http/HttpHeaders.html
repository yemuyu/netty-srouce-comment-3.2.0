415     /**
416      * Returns {@code true} if and only if the connection can remain open and
417      * thus 'kept alive'.  This methods respects the value of the
418      * {@code "Connection"} header first and then the return value of
419      * {@link HttpVersion#isKeepAliveDefault()}.
420      */
421     public static boolean isKeepAlive(HttpMessage message) {
422         String connection = message.getHeader(Names.CONNECTION);
423         if (Values.CLOSE.equalsIgnoreCase(connection)) {
424             return false;
425         }
426 
427         if (message.getProtocolVersion().isKeepAliveDefault()) {
428             return !Values.CLOSE.equalsIgnoreCase(connection);
429         } else {
430             return Values.KEEP_ALIVE.equalsIgnoreCase(connection);
431         }
432     }
433 
434     /**
435      * Sets the value of the {@code "Connection"} header depending on the
436      * protocol version of the specified message.  This method sets or removes
437      * the {@code "Connection"} header depending on what the default keep alive
438      * mode of the message's protocol version is, as specified by
439      * {@link HttpVersion#isKeepAliveDefault()}.
440      * <ul>
441      * <li>If the connection is kept alive by default:
442      *     <ul>
443      *     <li>set to {@code "close"} if {@code keepAlive} is {@code false}.</li>
444      *     <li>remove otherwise.</li>
445      *     </ul></li>
446      * <li>If the connection is closed by default:
447      *     <ul>
448      *     <li>set to {@code "keep-alive"} if {@code keepAlive} is {@code true}.</li>
449      *     <li>remove otherwise.</li>
450      *     </ul></li>
451      * </ul>
452      */
453     public static void setKeepAlive(HttpMessage message, boolean keepAlive) {
454         if (message.getProtocolVersion().isKeepAliveDefault()) {
455             if (keepAlive) {
456                 message.removeHeader(Names.CONNECTION);
457             } else {
458                 message.setHeader(Names.CONNECTION, Values.CLOSE);
459             }
460         } else {
461             if (keepAlive) {
462                 message.setHeader(Names.CONNECTION, Values.KEEP_ALIVE);
463             } else {
464                 message.removeHeader(Names.CONNECTION);
465             }
466         }
467     }

639     private static final int BUCKET_SIZE = 17;

641     private static int hash(String name) {
642         int h = 0;
643         for (int i = name.length() - 1; i >= 0; i --) {
644             char c = name.charAt(i);
645             if (c >= 'A' && c <= 'Z') {
646                 c += 32;
647             }
648             h = 31 * h + c;
649         }
650 
651         if (h > 0) {
652             return h;
653         } else if (h == Integer.MIN_VALUE) {
654             return Integer.MAX_VALUE;
655         } else {
656             return -h;
657         }
658     }
659 
660     private static boolean eq(String name1, String name2) {
661         int nameLen = name1.length();
662         if (nameLen != name2.length()) {
663             return false;
664         }
665 
666         for (int i = nameLen - 1; i >= 0; i --) {
667             char c1 = name1.charAt(i);
668             char c2 = name2.charAt(i);
669             if (c1 != c2) {
670                 if (c1 >= 'A' && c1 <= 'Z') {
671                     c1 += 32;
672                 }
673                 if (c2 >= 'A' && c2 <= 'Z') {
674                     c2 += 32;
675                 }
676                 if (c1 != c2) {
677                     return false;
678                 }
679             }
680         }
681         return true;
682     }

684     private static int index(int hash) {
685         return hash % BUCKET_SIZE;

688     private final Entry[] entries = new Entry[BUCKET_SIZE];
689     private final Entry head = new Entry(-1, null, null);

699     void addHeader(final String name, final Object value) {
700         validateHeaderName(name);
701         String strVal = toString(value);
702         HttpCodecUtil.validateHeaderValue(strVal);
703         int h = hash(name);
704         int i = index(h);
705         addHeader0(h, i, name, strVal);
706     }
707 
708     private void addHeader0(int h, int i, final String name, final String value) {
709         // Update the hash table.
710         Entry e = entries[i];
711         Entry newEntry;
712         entries[i] = newEntry = new Entry(h, name, value);
713         newEntry.next = e;
714 
715         // Update the linked list.
716         newEntry.addBefore(head);
717     }
718 
719     void removeHeader(final String name) {
720         if (name == null) {
721             throw new NullPointerException("name");
722         }
723         int h = hash(name);
724         int i = index(h);
725         removeHeader0(h, i, name);
726     }
727 
728     private void removeHeader0(int h, int i, String name) {
729         Entry e = entries[i];
730         if (e == null) {
731             return;
732         }
733 
734         for (;;) {
735             if (e.hash == h && eq(name, e.key)) {
736                 e.remove();
737                 Entry next = e.next;
738                 if (next != null) {
739                     entries[i] = next;
740                     e = next;
741                 } else {
742                     entries[i] = null;
743                     return;
744                 }
745             } else {
746                 break;
747             }
748         }
749 
750         for (;;) {
751             Entry next = e.next;
752             if (next == null) {
753                 break;
754             }
755             if (next.hash == h && eq(name, next.key)) {
756                 e.next = next.next;
757                 next.remove();
758             } else {
759                 e = next;
760             }
761         }
762     }

764     void setHeader(final String name, final Object value) {
765         validateHeaderName(name);
766         String strVal = toString(value);
767         HttpCodecUtil.validateHeaderValue(strVal);
768         int h = hash(name);
769         int i = index(h);
770         removeHeader0(h, i, name);
771         addHeader0(h, i, name, strVal);
772     }

774     void setHeader(final String name, final Iterable<?> values) {
775         if (values == null) {
776             throw new NullPointerException("values");
777         }
778 
779         validateHeaderName(name);
780 
781         int h = hash(name);
782         int i = index(h);
783 
784         removeHeader0(h, i, name);
785         for (Object v: values) {
786             if (v == null) {
787                 break;
