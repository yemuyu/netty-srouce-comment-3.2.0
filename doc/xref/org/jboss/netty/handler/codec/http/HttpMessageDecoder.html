28  /**
29   * Decodes {@link ChannelBuffer}s into {@link HttpMessage}s and
30   * {@link HttpChunk}s.
31   *
32   * <h3>Parameters that prevents excessive memory consumption</h3>
33   * <table border="1">
34   * <tr>
35   * <th>Name</th><th>Meaning</th>
36   * </tr>
37   * <tr>
38   * <td>{@code maxInitialLineLength}</td>
39   * <td>The maximum length of the initial line
40   *     (e.g. {@code "GET / HTTP/1.0"} or {@code "HTTP/1.0 200 OK"})
41   *     If the length of the initial line exceeds this value, a
42   *     {@link TooLongFrameException} will be raised.</td>
43   * </tr>
44   * <tr>
45   * <td>{@code maxHeaderSize}</td>
46   * <td>The maximum length of all headers.  If the sum of the length of each
47   *     header exceeds this value, a {@link TooLongFrameException} will be raised.</td>
48   * </tr>
49   * <tr>
50   * <td>{@code maxChunkSize}</td>
51   * <td>The maximum length of the content or each chunk.  If the content length
52   *     (or the length of each chunk) exceeds this value, the content or chunk
53   *     will be split into multiple {@link HttpChunk}s whose length is
54   *     {@code maxChunkSize} at maximum.</td>
55   * </tr>
56   * </table>
57   *
58   * <h3>Chunked Content</h3>
59   *
60   * If the content of an HTTP message is greater than {@code maxChunkSize} or
61   * the transfer encoding of the HTTP message is 'chunked', this decoder
62   * generates one {@link HttpMessage} instance and its following
63   * {@link HttpChunk}s per single HTTP message to avoid excessive memory
64   * consumption. For example, the following HTTP message:
65   * <pre>
66   * GET / HTTP/1.1
67   * Transfer-Encoding: chunked
68   *
69   * 1a
70   * abcdefghijklmnopqrstuvwxyz
71   * 10
72   * 1234567890abcdef
73   * 0
74   * Content-MD5: ...
75   * <i>[blank line]</i>
76   * </pre>
77   * triggers {@link HttpRequestDecoder} to generate 4 objects:
78   * <ol>
79   * <li>An {@link HttpRequest} whose {@link HttpMessage#isChunked() chunked}
80   *     property is {@code true},</li>
81   * <li>The first {@link HttpChunk} whose content is {@code 'abcdefghijklmnopqrstuvwxyz'},</li>
82   * <li>The second {@link HttpChunk} whose content is {@code '1234567890abcdef'}, and</li>
83   * <li>An {@link HttpChunkTrailer} which marks the end of the content.</li>
84   * </ol>
85   *
86   * If you prefer not to handle {@link HttpChunk}s by yourself for your
87   * convenience, insert {@link HttpChunkAggregator} after this decoder in the
88   * {@link ChannelPipeline}.  However, please note that your server might not
89   * be as memory efficient as without the aggregator.
90   *
91   * <h3>Extensibility</h3>
92   *
93   * Please note that this decoder is designed to be extended to implement
94   * a protocol derived from HTTP, such as
95   * <a href="http://en.wikipedia.org/wiki/Real_Time_Streaming_Protocol">RTSP</a> and
96   * <a href="http://en.wikipedia.org/wiki/Internet_Content_Adaptation_Protocol">ICAP</a>.
97   * To implement the decoder of such a derived protocol, extend this class and
98   * implement all abstract methods properly.
99   *
100  * @author <a href="http://www.jboss.org/netty/">The Netty Project</a>
101  * @author Andy Taylor (andy.taylor@jboss.org)
102  * @author <a href="http://gleamynode.net/">Trustin Lee</a>
103  * @version $Rev$, $Date$
104  *
105  * @apiviz.landmark
106  */
107 public abstract class HttpMessageDecoder extends ReplayingDecoder<HttpMessageDecoder.State> {

130         READ_HEADER,

154     protected HttpMessageDecoder(
155             int maxInitialLineLength, int maxHeaderSize, int maxChunkSize) {
156 
157         super(State.SKIP_CONTROL_CHARS, true);
158 
159         if (maxInitialLineLength <= 0) {
160             throw new IllegalArgumentException(
161                     "maxInitialLineLength must be a positive integer: " +
162                     maxInitialLineLength);
163         }
164         if (maxHeaderSize <= 0) {
165             throw new IllegalArgumentException(
166                     "maxHeaderSize must be a positive integer: " +
167                     maxChunkSize);
168         }
169         if (maxChunkSize < 0) {
170             throw new IllegalArgumentException(
171                     "maxChunkSize must be a positive integer: " +
172                     maxChunkSize);
173         }
174         this.maxInitialLineLength = maxInitialLineLength;
175         this.maxHeaderSize = maxHeaderSize;
176         this.maxChunkSize = maxChunkSize;
177     }

179     @Override
180     protected Object decode(ChannelHandlerContext ctx, Channel channel, ChannelBuffer buffer, State state) throws Exception {
181         switch (state) {
182         case SKIP_CONTROL_CHARS: {
183             try {
184                 skipControlCharacters(buffer);
185                 checkpoint(State.READ_INITIAL);
186             } finally {
187                 checkpoint();
188             }
189         }
190         case READ_INITIAL: {
191             String[] initialLine = splitInitialLine(readLine(buffer, maxInitialLineLength));
192             if (initialLine.length < 3) {
193                 // Invalid initial line - ignore.
194                 checkpoint(State.SKIP_CONTROL_CHARS);
195                 return null;
196             }
197 
198             message = createMessage(initialLine);
199             checkpoint(State.READ_HEADER);
200         }
201         case READ_HEADER: {
202             State nextState = readHeaders(buffer);
203             checkpoint(nextState);
204             if (nextState == State.READ_CHUNK_SIZE) {
205                 // Chunked encoding
206                 message.setChunked(true);
207                 // Generate HttpMessage first.  HttpChunks will follow.
208                 return message;
209             } else if (nextState == State.SKIP_CONTROL_CHARS) {
210                 // No content is expected.
211                 // Remove the headers which are not supposed to be present not
212                 // to confuse subsequent handlers.
213                 message.removeHeader(HttpHeaders.Names.TRANSFER_ENCODING);
214                 return message;
215             } else {
216                 long contentLength = HttpHeaders.getContentLength(message, -1);
217                 if (contentLength == 0 || contentLength == -1 && isDecodingRequest()) {
218                     content = ChannelBuffers.EMPTY_BUFFER;
219                     return reset();
220                 }
221 
222                 switch (nextState) {
223                 case READ_FIXED_LENGTH_CONTENT:
224                     if (contentLength > maxChunkSize) {
225                         // Generate HttpMessage first.  HttpChunks will follow.
226                         checkpoint(State.READ_FIXED_LENGTH_CONTENT_AS_CHUNKS);
227                         message.setChunked(true);
228                         // chunkSize will be decreased as the READ_FIXED_LENGTH_CONTENT_AS_CHUNKS
229                         // state reads data chunk by chunk.
230                         chunkSize = HttpHeaders.getContentLength(message, -1);
231                         return message;
232                     }
233                     break;
234                 case READ_VARIABLE_LENGTH_CONTENT:
235                     if (buffer.readableBytes() > maxChunkSize) {
236                         // Generate HttpMessage first.  HttpChunks will follow.
237                         checkpoint(State.READ_VARIABLE_LENGTH_CONTENT_AS_CHUNKS);
238                         message.setChunked(true);
239                         return message;
240                     }
241                     break;
242                 }
243             }
244             // We return null here, this forces decode to be called again where we will decode the content
245             return null;
246         }
247         case READ_VARIABLE_LENGTH_CONTENT: {
248             if (content == null) {
249                 content = ChannelBuffers.dynamicBuffer(channel.getConfig().getBufferFactory());
250             }
251             //this will cause a replay error until the channel is closed where this will read what's left in the buffer
252             content.writeBytes(buffer.readBytes(buffer.readableBytes()));
253             return reset();
254         }
255         case READ_VARIABLE_LENGTH_CONTENT_AS_CHUNKS: {
256             // Keep reading data as a chunk until the end of connection is reached.
257             int chunkSize = Math.min(maxChunkSize, buffer.readableBytes());
258             HttpChunk chunk = new DefaultHttpChunk(buffer.readBytes(chunkSize));
259 
260             if (!buffer.readable()) {
261                 // Reached to the end of the connection.
262                 reset();
263                 if (!chunk.isLast()) {
264                     // Append the last chunk.
265                     return new Object[] { chunk, HttpChunk.LAST_CHUNK };
266                 }
267             }
268             return chunk;
269         }
270         case READ_FIXED_LENGTH_CONTENT: {
271             //we have a content-length so we just read the correct number of bytes
272             readFixedLengthContent(buffer);
273             return reset();
274         }
275         case READ_FIXED_LENGTH_CONTENT_AS_CHUNKS: {
276             long chunkSize = this.chunkSize;
277             HttpChunk chunk;
278             if (chunkSize > maxChunkSize) {
279                 chunk = new DefaultHttpChunk(buffer.readBytes(maxChunkSize));
280                 chunkSize -= maxChunkSize;
281             } else {
282                 assert chunkSize <= Integer.MAX_VALUE;
283                 chunk = new DefaultHttpChunk(buffer.readBytes((int) chunkSize));
284                 chunkSize = 0;
285             }
286             this.chunkSize = chunkSize;
287 
288             if (chunkSize == 0) {
289                 // Read all content.
290                 reset();
291                 if (!chunk.isLast()) {
292                     // Append the last chunk.
293                     return new Object[] { chunk, HttpChunk.LAST_CHUNK };
294                 }
295             }
296             return chunk;
297         }
298         /**
299          * everything else after this point takes care of reading chunked content. basically, read chunk size,
300          * read chunk, read and ignore the CRLF and repeat until 0
301          */
302         case READ_CHUNK_SIZE: {
303             String line = readLine(buffer, maxInitialLineLength);
304             int chunkSize = getChunkSize(line);
305             this.chunkSize = chunkSize;
306             if (chunkSize == 0) {
307                 checkpoint(State.READ_CHUNK_FOOTER);
308                 return null;
309             } else if (chunkSize > maxChunkSize) {
310                 // A chunk is too large. Split them into multiple chunks again.
311                 checkpoint(State.READ_CHUNKED_CONTENT_AS_CHUNKS);
312             } else {
313                 checkpoint(State.READ_CHUNKED_CONTENT);
314             }
315         }
316         case READ_CHUNKED_CONTENT: {
317             assert chunkSize <= Integer.MAX_VALUE;
318             HttpChunk chunk = new DefaultHttpChunk(buffer.readBytes((int) chunkSize));
319             checkpoint(State.READ_CHUNK_DELIMITER);
320             return chunk;
321         }
322         case READ_CHUNKED_CONTENT_AS_CHUNKS: {
323             long chunkSize = this.chunkSize;
324             HttpChunk chunk;
325             if (chunkSize > maxChunkSize) {
326                 chunk = new DefaultHttpChunk(buffer.readBytes(maxChunkSize));
327                 chunkSize -= maxChunkSize;
328             } else {
329                 assert chunkSize <= Integer.MAX_VALUE;
330                 chunk = new DefaultHttpChunk(buffer.readBytes((int) chunkSize));
331                 chunkSize = 0;
332             }
333             this.chunkSize = chunkSize;
334 
335             if (chunkSize == 0) {
336                 // Read all content.
337                 checkpoint(State.READ_CHUNK_DELIMITER);
338             }
339 
340             if (!chunk.isLast()) {
341                 return chunk;
342             }
343         }
344         case READ_CHUNK_DELIMITER: {
345             for (;;) {
346                 byte next = buffer.readByte();
347                 if (next == HttpCodecUtil.CR) {
348                     if (buffer.readByte() == HttpCodecUtil.LF) {
349                         checkpoint(State.READ_CHUNK_SIZE);
350                         return null;
351                     }
352                 } else if (next == HttpCodecUtil.LF) {
353                     checkpoint(State.READ_CHUNK_SIZE);
354                     return null;
355                 }
356             }
357         }
358         case READ_CHUNK_FOOTER: {
359             HttpChunkTrailer trailer = readTrailingHeaders(buffer);
360             if (maxChunkSize == 0) {
361                 // Chunked encoding disabled.
362                 return reset();
363             } else {
364                 reset();
365                 // The last chunk, which is empty
366                 return trailer;
367             }
368         }
369         default: {
370             throw new Error("Shouldn't reach here.");
371         }
372 
373         }
374     }

376     protected boolean isContentAlwaysEmpty(HttpMessage msg) {
377         if (msg instanceof HttpResponse) {
378             HttpResponse res = (HttpResponse) msg;
379             int code = res.getStatus().getCode();
380             if (code < 200) {
381                 return true;
382             }
383             switch (code) {
384             case 204: case 205: case 304:
385                 return true;
386             }
387         }
388         return false;
389     }

391     private Object reset() {
392         HttpMessage message = this.message;
393         ChannelBuffer content = this.content;
394 
395         if (content != null) {
396             message.setContent(content);
397             this.content = null;
398         }
399         this.message = null;
400 
401         checkpoint(State.SKIP_CONTROL_CHARS);
402         return message;
403     }

405     private void skipControlCharacters(ChannelBuffer buffer) {
406         for (;;) {
407             char c = (char) buffer.readUnsignedByte();
408             if (!Character.isISOControl(c) &&
409                 !Character.isWhitespace(c)) {
410                 buffer.readerIndex(buffer.readerIndex() - 1);
411                 break;
412             }
413         }
414     }

416     private void readFixedLengthContent(ChannelBuffer buffer) {
417         long length = HttpHeaders.getContentLength(message, -1);
418         assert length <= Integer.MAX_VALUE;
419 
420         if (content == null) {
421             content = buffer.readBytes((int) length);
422         } else {
423             content.writeBytes(buffer.readBytes((int) length));
424         }
425     }

427     private State readHeaders(ChannelBuffer buffer) throws TooLongFrameException {
428         headerSize = 0;
429         final HttpMessage message = this.message;
430         String line = readHeader(buffer);
431         String lastHeader = null;
432         if (line.length() != 0) {
433             message.clearHeaders();
434             do {
435                 char firstChar = line.charAt(0);
436                 if (lastHeader != null && (firstChar == ' ' || firstChar == '\t')) {
437                     List<String> current = message.getHeaders(lastHeader);
438                     int lastPos = current.size() - 1;
439                     String newString = current.get(lastPos) + line.trim();
440                     current.set(lastPos, newString);
441                 } else {
442                     String[] header = splitHeader(line);
443                     message.addHeader(header[0], header[1]);
444                     lastHeader = header[0];
445                 }
446 
447                 line = readHeader(buffer);
448             } while (line.length() != 0);
449         }
450 
451         State nextState;
452 
453         if (isContentAlwaysEmpty(message)) {
454             nextState = State.SKIP_CONTROL_CHARS;
455         } else if (message.isChunked()) {
456             // HttpMessage.isChunked() returns true when either:
457             // 1) HttpMessage.setChunked(true) was called or
458             // 2) 'Transfer-Encoding' is 'chunked'.
459             // Because this decoder did not call HttpMessage.setChunked(true)
460             // yet, HttpMessage.isChunked() should return true only when
461             // 'Transfer-Encoding' is 'chunked'.
462             nextState = State.READ_CHUNK_SIZE;
463         } else if (HttpHeaders.getContentLength(message, -1) >= 0) {
464             nextState = State.READ_FIXED_LENGTH_CONTENT;
465         } else {
466             nextState = State.READ_VARIABLE_LENGTH_CONTENT;
467         }
468         return nextState;
469     }

471     private HttpChunkTrailer readTrailingHeaders(ChannelBuffer buffer) throws TooLongFrameException {
472         headerSize = 0;
473         String line = readHeader(buffer);
474         String lastHeader = null;
475         if (line.length() != 0) {
476             HttpChunkTrailer trailer = new DefaultHttpChunkTrailer();
477             do {
478                 char firstChar = line.charAt(0);
479                 if (lastHeader != null && (firstChar == ' ' || firstChar == '\t')) {
480                     List<String> current = trailer.getHeaders(lastHeader);
481                     if (current.size() != 0) {
482                         int lastPos = current.size() - 1;
483                         String newString = current.get(lastPos) + line.trim();
484                         current.set(lastPos, newString);
485                     } else {
486                         // Content-Length, Transfer-Encoding, or Trailer
487                     }
488                 } else {
489                     String[] header = splitHeader(line);
490                     String name = header[0];
491                     if (!name.equalsIgnoreCase(HttpHeaders.Names.CONTENT_LENGTH) &&
492                         !name.equalsIgnoreCase(HttpHeaders.Names.TRANSFER_ENCODING) &&
493                         !name.equalsIgnoreCase(HttpHeaders.Names.TRAILER)) {
494                         trailer.addHeader(name, header[1]);
495                     }
496                     lastHeader = name;
497                 }
498 
499                 line = readHeader(buffer);
500             } while (line.length() != 0);
501 
502             return trailer;
503         }
504 
505         return HttpChunk.LAST_CHUNK;
506     }

508     private String readHeader(ChannelBuffer buffer) throws TooLongFrameException {
509         StringBuilder sb = new StringBuilder(64);
510         int headerSize = this.headerSize;
511 
512         loop:
513         for (;;) {
514             char nextByte = (char) buffer.readByte();
515             headerSize ++;
516 
517             switch (nextByte) {
518             case HttpCodecUtil.CR:
519                 nextByte = (char) buffer.readByte();
520                 headerSize ++;
521                 if (nextByte == HttpCodecUtil.LF) {
522                     break loop;
523                 }
524                 break;
525             case HttpCodecUtil.LF:
526                 break loop;
527             }
528 
529             // Abort decoding if the header part is too large.
530             if (headerSize >= maxHeaderSize) {
531                 throw new TooLongFrameException(
532                         "HTTP header is larger than " +
533                         maxHeaderSize + " bytes.");
534 
535             }
536 
537             sb.append(nextByte);
538         }
539 
540         this.headerSize = headerSize;
541         return sb.toString();
542     }

547     private int getChunkSize(String hex) {
548         hex = hex.trim();
549         for (int i = 0; i < hex.length(); i ++) {
550             char c = hex.charAt(i);
551             if (c == ';' || Character.isWhitespace(c) || Character.isISOControl(c)) {
552                 hex = hex.substring(0, i);
553                 break;
554             }
555         }
556 
557         return Integer.parseInt(hex, 16);
558     }

560     private String readLine(ChannelBuffer buffer, int maxLineLength) throws TooLongFrameException {
561         StringBuilder sb = new StringBuilder(64);
562         int lineLength = 0;
563         while (true) {
564             byte nextByte = buffer.readByte();
565             if (nextByte == HttpCodecUtil.CR) {
566                 nextByte = buffer.readByte();
567                 if (nextByte == HttpCodecUtil.LF) {
568                     return sb.toString();
569                 }
570             }
571             else if (nextByte == HttpCodecUtil.LF) {
572                 return sb.toString();
573             }
574             else {
575                 if (lineLength >= maxLineLength) {
576                     throw new TooLongFrameException(
577                             "An HTTP line is larger than " + maxLineLength +
578                             " bytes.");
579                 }
580                 lineLength ++;
581                 sb.append((char) nextByte);
582             }
583         }
584     }

586     private String[] splitInitialLine(String sb) {
587         int aStart;
588         int aEnd;
589         int bStart;
590         int bEnd;
591         int cStart;
592         int cEnd;
593 
594         aStart = findNonWhitespace(sb, 0);
595         aEnd = findWhitespace(sb, aStart);
596 
597         bStart = findNonWhitespace(sb, aEnd);
598         bEnd = findWhitespace(sb, bStart);
599 
600         cStart = findNonWhitespace(sb, bEnd);
601         cEnd = findEndOfString(sb);
602 
603         return new String[] {
604                 sb.substring(aStart, aEnd),
605                 sb.substring(bStart, bEnd),
606                 cStart < cEnd? sb.substring(cStart, cEnd) : "" };
607     }

609     private String[] splitHeader(String sb) {
610         final int length = sb.length();
611         int nameStart;
612         int nameEnd;
613         int colonEnd;
614         int valueStart;
615         int valueEnd;
616 
617         nameStart = findNonWhitespace(sb, 0);
618         for (nameEnd = nameStart; nameEnd < length; nameEnd ++) {
619             char ch = sb.charAt(nameEnd);
620             if (ch == ':' || Character.isWhitespace(ch)) {
621                 break;
622             }
623         }
624 
625         for (colonEnd = nameEnd; colonEnd < length; colonEnd ++) {
626             if (sb.charAt(colonEnd) == ':') {
627                 colonEnd ++;
628                 break;
629             }
630         }
631 
632         valueStart = findNonWhitespace(sb, colonEnd);
633         if (valueStart == length) {
634             return new String[] {
635                     sb.substring(nameStart, nameEnd),
636                     ""
637             };
638         }
639 
640         valueEnd = findEndOfString(sb);
641         return new String[] {
642                 sb.substring(nameStart, nameEnd),
643                 sb.substring(valueStart, valueEnd)
644         };
645     }

647     private int findNonWhitespace(String sb, int offset) {
648         int result;
649         for (result = offset; result < sb.length(); result ++) {
650             if (!Character.isWhitespace(sb.charAt(result))) {
651                 break;
652             }
653         }
654         return result;
655     }
656 
657     private int findWhitespace(String sb, int offset) {
658         int result;
659         for (result = offset; result < sb.length(); result ++) {
660             if (Character.isWhitespace(sb.charAt(result))) {
661                 break;
662             }
663         }
664         return result;
665     }
666 
667     private int findEndOfString(String sb) {
668         int result;
669         for (result = sb.length(); result > 0; result --) {
670             if (!Character.isWhitespace(sb.charAt(result - 1))) {
671                 break;
672             }
673         }
674         return result;
675     }
