16  package org.jboss.netty.handler.codec.http;
17  
18  import org.jboss.netty.buffer.ChannelBuffer;
19  import org.jboss.netty.handler.codec.compression.ZlibDecoder;
20  import org.jboss.netty.handler.codec.compression.ZlibWrapper;
21  import org.jboss.netty.handler.codec.embedder.DecoderEmbedder;
22  
23  /**
24   * Decompresses an {@link HttpMessage} and an {@link HttpChunk} compressed in
25   * {@code gzip} or {@code deflate} encoding.  For more information on how this
26   * handler modifies the message, please refer to {@link HttpContentDecoder}.
27   *
28   * @author <a href="http://www.jboss.org/netty/">The Netty Project</a>
29   * @author <a href="http://gleamynode.net/">Trustin Lee</a>
30   * @version $Rev$, $Date$
31   */
32  public class HttpContentDecompressor extends HttpContentDecoder {
33      @Override
34      protected DecoderEmbedder<ChannelBuffer> newContentDecoder(String contentEncoding) throws Exception {
35          if ("gzip".equalsIgnoreCase(contentEncoding) || "x-gzip".equalsIgnoreCase(contentEncoding)) {
36              return new DecoderEmbedder<ChannelBuffer>(new ZlibDecoder(ZlibWrapper.GZIP));
37          } else if ("deflate".equalsIgnoreCase(contentEncoding) || "x-deflate".equalsIgnoreCase(contentEncoding)) {
38              return new DecoderEmbedder<ChannelBuffer>(new ZlibDecoder(ZlibWrapper.ZLIB));
39          }
40  
41          // 'identity' or unsupported
42          return null;
43      }
44  }
