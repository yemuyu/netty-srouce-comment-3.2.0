55  /**
56   * Adds <a href="http://en.wikipedia.org/wiki/Transport_Layer_Security">SSL
57   * &middot; TLS</a> and StartTLS support to a {@link Channel}.  Please refer
58   * to the <strong>"SecureChat"</strong> example in the distribution or the web
59   * site for the detailed usage.
60   *
61   * <h3>Beginning the handshake</h3>
62   * <p>
63   * You must make sure not to write a message while the
64   * {@linkplain #handshake() handshake} is in progress unless you are
65   * renegotiating.  You will be notified by the {@link ChannelFuture} which is
66   * returned by the {@link #handshake()} method when the handshake
67   * process succeeds or fails.
68   *
69   * <h3>Renegotiation</h3>
70   * <p>
71   * TLS renegotiation has been disabled by default due to a known security issue,
72   * <a href="http://cve.mitre.org/cgi-bin/cvename.cgi?name=CVE-2009-3555">CVE-2009-3555</a>.
73   * You can re-enable renegotiation by calling {@link #setEnableRenegotiation(boolean)}
74   * with {@code true} at your own risk.
75   * <p>
76   * If {@link #isEnableRenegotiation() enableRenegotiation} is {@code true} and
77   * the initial handshake has been done successfully, you can call
78   * {@link #handshake()} to trigger the renegotiation.
79   * <p>
80   * If {@link #isEnableRenegotiation() enableRenegotiation} is {@code false},
81   * an attempt to trigger renegotiation will result in the connection closure.
82   *
83   * <h3>Closing the session</h3>
84   * <p>
85   * To close the SSL session, the {@link #close()} method should be
86   * called to send the {@code close_notify} message to the remote peer.  One
87   * exception is when you close the {@link Channel} - {@link SslHandler}
88   * intercepts the close request and send the {@code close_notify} message
89   * before the channel closure automatically.  Once the SSL session is closed,
90   * it is not reusable, and consequently you should create a new
91   * {@link SslHandler} with a new {@link SSLEngine} as explained in the
92   * following section.
93   *
94   * <h3>Restarting the session</h3>
95   * <p>
96   * To restart the SSL session, you must remove the existing closed
97   * {@link SslHandler} from the {@link ChannelPipeline}, insert a new
98   * {@link SslHandler} with a new {@link SSLEngine} into the pipeline,
99   * and start the handshake process as described in the first section.
100  *
101  * <h3>Implementing StartTLS</h3>
102  * <p>
103  * <a href="http://en.wikipedia.org/wiki/STARTTLS">StartTLS</a> is the
104  * communication pattern that secures the wire in the middle of the plaintext
105  * connection.  Please note that it is different from SSL &middot; TLS, that
106  * secures the wire from the beginning of the connection.  Typically, StartTLS
107  * is composed of three steps:
108  * <ol>
109  * <li>Client sends a StartTLS request to server.</li>
110  * <li>Server sends a StartTLS response to client.</li>
111  * <li>Client begins SSL handshake.</li>
112  * </ol>
113  * If you implement a server, you need to:
114  * <ol>
115  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
116  *     to {@code true},</li>
117  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
118  * <li>write a StartTLS response.</li>
119  * </ol>
120  * Please note that you must insert {@link SslHandler} <em>before</em> sending
121  * the StartTLS response.  Otherwise the client can send begin SSL handshake
122  * before {@link SslHandler} is inserted to the {@link ChannelPipeline}, causing
123  * data corruption.
124  * <p>
125  * The client-side implementation is much simpler.
126  * <ol>
127  * <li>Write a StartTLS request,</li>
128  * <li>wait for the StartTLS response,</li>
129  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
130  *     to {@code false},</li>
131  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
132  * <li>Initiate SSL handshake by calling {@link SslHandler#handshake()}.</li>
133  * </ol>
134  *
135  * @author <a href="http://www.jboss.org/netty/">The Netty Project</a>
136  * @author <a href="http://gleamynode.net/">Trustin Lee</a>
137  *
138  * @version $Rev$, $Date$
139  *
140  * @apiviz.landmark
141  * @apiviz.uses org.jboss.netty.handler.ssl.SslBufferPool
142  */

317     /**
318      * Starts an SSL / TLS handshake for the specified channel.
319      *
320      * @return a {@link ChannelFuture} which is notified when the handshake
321      *         succeeds or fails.
322      */
323     public ChannelFuture handshake() {
324         if (handshaken && !isEnableRenegotiation()) {
325             throw new IllegalStateException("renegotiation disabled");
326         }
327 
328         ChannelHandlerContext ctx = this.ctx;
329         Channel channel = ctx.getChannel();
330         ChannelFuture handshakeFuture;
331         synchronized (handshakeLock) {
332             if (handshaking) {
333                 return this.handshakeFuture;
334             } else {
335                 handshaking = true;
336                 try {
337                     engine.beginHandshake();
338                     runDelegatedTasks();
339                     handshakeFuture = this.handshakeFuture = future(channel);
340                 } catch (SSLException e) {
341                     handshakeFuture = this.handshakeFuture = failedFuture(channel, e);
342                 }
343             }
344         }
345 
346         try {
347             wrapNonAppData(ctx, channel);
348         } catch (SSLException e) {
349             handshakeFuture.setFailure(e);
350         }
351         return handshakeFuture;
352     }

399     public void handleDownstream(
400             final ChannelHandlerContext context, final ChannelEvent evt) throws Exception {
401         if (evt instanceof ChannelStateEvent) {
402             ChannelStateEvent e = (ChannelStateEvent) evt;
403             switch (e.getState()) {
404             case OPEN:
405             case CONNECTED:
406             case BOUND:
407                 if (Boolean.FALSE.equals(e.getValue()) || e.getValue() == null) {
408                     closeOutboundAndChannel(context, e);
409                     return;
410                 }
411             }
412         }
413         if (!(evt instanceof MessageEvent)) {
414             context.sendDownstream(evt);
415             return;
416         }
417 
418         MessageEvent e = (MessageEvent) evt;
419         if (!(e.getMessage() instanceof ChannelBuffer)) {
420             context.sendDownstream(evt);
421             return;
422         }
423 
424         // Do not encrypt the first write request if this handler is
425         // created with startTLS flag turned on.
426         if (startTls && sentFirstMessage.compareAndSet(false, true)) {
427             context.sendDownstream(evt);
428             return;
429         }
430 
431         // Otherwise, all messages are encrypted.
432         ChannelBuffer msg = (ChannelBuffer) e.getMessage();
433         PendingWrite pendingWrite;
434 
435         if (msg.readable()) {
436             pendingWrite = new PendingWrite(evt.getFuture(), msg.toByteBuffer(msg.readerIndex(), msg.readableBytes()));
437         } else {
438             pendingWrite = new PendingWrite(evt.getFuture(), null);
439         }
440         synchronized (pendingUnencryptedWrites) {
441             boolean offered = pendingUnencryptedWrites.offer(pendingWrite);
442             assert offered;
443         }
444 
445         wrap(context, evt.getChannel());
446     }

448     @Override
449     public void channelDisconnected(ChannelHandlerContext ctx,
450             ChannelStateEvent e) throws Exception {
451 
452         // Make sure the handshake future is notified when a connection has
453         // been closed during handshake.
454         synchronized (handshakeLock) {
455             if (handshaking) {
456                 handshakeFuture.setFailure(new ClosedChannelException());
457             }
458         }
459 
460         try {
461             super.channelDisconnected(ctx, e);
462         } finally {
463             unwrap(ctx, e.getChannel(), ChannelBuffers.EMPTY_BUFFER, 0, 0);
464             engine.closeOutbound();
465             if (!sentCloseNotify.get() && handshaken) {
466                 try {
467                     engine.closeInbound();
468                 } catch (SSLException ex) {
469                     logger.debug("Failed to clean up SSLEngine.", ex);
470                 }
471             }
472         }
473     }

475     @Override
476     public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e)
477             throws Exception {
478 
479         Throwable cause = e.getCause();
480         if (cause instanceof IOException) {
481             if (cause instanceof ClosedChannelException) {
482                 synchronized (ignoreClosedChannelExceptionLock) {
483                     if (ignoreClosedChannelException > 0) {
484                         ignoreClosedChannelException --;
485                         logger.debug(
486                                 "Swallowing an exception raised while " +
487                                 "writing non-app data", cause);
488                         return;
489                     }
490                 }
491             } else if (engine.isOutboundDone()) {
492                 String message = String.valueOf(cause.getMessage()).toLowerCase();
493                 if (IGNORABLE_ERROR_MESSAGE.matcher(message).matches()) {
494                     // It is safe to ignore the 'connection reset by peer' or
495                     // 'broken pipe' error after sending closure_notify.
496                     logger.debug(
497                             "Swallowing a 'connection reset by peer / " +
498                             "broken pipe' error occurred while writing " +
499                             "'closure_notify'", cause);
500 
501                     // Close the connection explicitly just in case the transport
502                     // did not close the connection automatically.
503                     Channels.close(ctx, succeededFuture(e.getChannel()));
504                     return;
505                 }
506             }
507         }
508 
509         ctx.sendUpstream(e);
510     }

512     @Override
513     protected Object decode(
514             final ChannelHandlerContext ctx, Channel channel, ChannelBuffer buffer) throws Exception {
515 
516         if (buffer.readableBytes() < 5) {
517             return null;
518         }
519 
520         int packetLength = 0;
521 
522         // SSLv3 or TLS - Check ContentType
523         boolean tls;
524         switch (buffer.getUnsignedByte(buffer.readerIndex())) {
525         case 20:  // change_cipher_spec
526         case 21:  // alert
527         case 22:  // handshake
528         case 23:  // application_data
529             tls = true;
530             break;
531         default:
532             // SSLv2 or bad data
533             tls = false;
534         }
535 
536         if (tls) {
537             // SSLv3 or TLS - Check ProtocolVersion
538             int majorVersion = buffer.getUnsignedByte(buffer.readerIndex() + 1);
539             if (majorVersion >= 3 && majorVersion < 10) {
540                 // SSLv3 or TLS
541                 packetLength = (buffer.getShort(buffer.readerIndex() + 3) & 0xFFFF) + 5;
542                 if (packetLength <= 5) {
543                     // Neither SSLv2 or TLSv1 (i.e. SSLv2 or bad data)
544                     tls = false;
545                 }
546             } else {
547                 // Neither SSLv2 or TLSv1 (i.e. SSLv2 or bad data)
548                 tls = false;
549             }
550         }
551 
552         if (!tls) {
553             // SSLv2 or bad data - Check the version
554             boolean sslv2 = true;
555             int headerLength = (buffer.getUnsignedByte(
556                     buffer.readerIndex()) & 0x80) != 0 ? 2 : 3;
557             int majorVersion = buffer.getUnsignedByte(
558                     buffer.readerIndex() + headerLength + 1);
559             if (majorVersion >= 2 && majorVersion < 10) {
560                 // SSLv2
561                 if (headerLength == 2) {
562                     packetLength = (buffer.getShort(buffer.readerIndex()) & 0x7FFF) + 2;
563                 } else {
564                     packetLength = (buffer.getShort(buffer.readerIndex()) & 0x3FFF) + 3;
565                 }
566                 if (packetLength <= headerLength) {
567                     sslv2 = false;
568                 }
569             } else {
570                 sslv2 = false;
571             }
572 
573             if (!sslv2) {
574                 // Bad data - discard the buffer and raise an exception.
575                 SSLException e = new SSLException(
576                         "not an SSL/TLS record: " + ChannelBuffers.hexDump(buffer));
577                 buffer.skipBytes(buffer.readableBytes());
578                 throw e;
579             }
580         }
581 
582         assert packetLength > 0;
583 
584         if (buffer.readableBytes() < packetLength) {
585             return null;
586         }
587 
588         // We advance the buffer's readerIndex before calling unwrap() because
589         // unwrap() can trigger FrameDecoder call decode(), this method, recursively.
590         // The recursive call results in decoding the same packet twice if
591         // the readerIndex is advanced *after* decode().
592         //
593         // Here's an example:
594         // 1) An SSL packet is received from the wire.
595         // 2) SslHandler.decode() deciphers the packet and calls the user code.
596         // 3) The user closes the channel in the same thread.
597         // 4) The same thread triggers a channelDisconnected() event.
598         // 5) FrameDecoder.cleanup() is called, and it calls SslHandler.decode().
599         // 6) SslHandler.decode() will feed the same packet with what was
600         //    deciphered at the step 2 again if the readerIndex was not advanced
601         //    before calling the user code.
602         final int packetOffset = buffer.readerIndex();
603         buffer.skipBytes(packetLength);
604         return unwrap(ctx, channel, buffer, packetOffset, packetLength);
605     }

607     private ChannelFuture wrap(ChannelHandlerContext context, Channel channel)
608             throws SSLException {
609 
610         ChannelFuture future = null;
611         ChannelBuffer msg;
612         ByteBuffer outNetBuf = bufferPool.acquire();
613         boolean success = true;
614         boolean offered = false;
615         boolean needsUnwrap = false;
616         try {
617             loop:
618             for (;;) {
619                 // Acquire a lock to make sure unencrypted data is polled
620                 // in order and their encrypted counterpart is offered in
621                 // order.
622                 synchronized (pendingUnencryptedWrites) {
623                     PendingWrite pendingWrite = pendingUnencryptedWrites.peek();
624                     if (pendingWrite == null) {
625                         break;
626                     }
627 
628                     ByteBuffer outAppBuf = pendingWrite.outAppBuf;
629                     if (outAppBuf == null) {
630                         // A write request with an empty buffer
631                         pendingUnencryptedWrites.remove();
632                         offerEncryptedWriteRequest(
633                                 new DownstreamMessageEvent(
634                                         channel, pendingWrite.future,
635                                         ChannelBuffers.EMPTY_BUFFER,
636                                         channel.getRemoteAddress()));
637                         offered = true;
638                     } else {
639                         SSLEngineResult result = null;
640                         try {
641                             synchronized (handshakeLock) {
642                                 result = engine.wrap(outAppBuf, outNetBuf);
643                             }
644                         } finally {
645                             if (!outAppBuf.hasRemaining()) {
646                                 pendingUnencryptedWrites.remove();
647                             }
648                         }
649 
650                         if (result.bytesProduced() > 0) {
651                             outNetBuf.flip();
652                             msg = ChannelBuffers.buffer(outNetBuf.remaining());
653                             msg.writeBytes(outNetBuf.array(), 0, msg.capacity());
654                             outNetBuf.clear();
655 
656                             if (pendingWrite.outAppBuf.hasRemaining()) {
657                                 // pendingWrite's future shouldn't be notified if
658                                 // only partial data is written.
659                                 future = succeededFuture(channel);
660                             } else {
661                                 future = pendingWrite.future;
662                             }
663 
664                             MessageEvent encryptedWrite = new DownstreamMessageEvent(
665                                     channel, future, msg, channel.getRemoteAddress());
666                             offerEncryptedWriteRequest(encryptedWrite);
667                             offered = true;
668                         } else {
669                             final HandshakeStatus handshakeStatus = result.getHandshakeStatus();
670                             handleRenegotiation(handshakeStatus);
671                             switch (handshakeStatus) {
672                             case NEED_WRAP:
673                                 if (outAppBuf.hasRemaining()) {
674                                     break;
675                                 } else {
676                                     break loop;
677                                 }
678                             case NEED_UNWRAP:
679                                 needsUnwrap = true;
680                                 break loop;
681                             case NEED_TASK:
682                                 runDelegatedTasks();
683                                 break;
684                             case FINISHED:
685                             case NOT_HANDSHAKING:
686                                 if (handshakeStatus == HandshakeStatus.FINISHED) {
687                                     setHandshakeSuccess(channel);
688                                 }
689                                 if (result.getStatus() == Status.CLOSED) {
690                                     success = false;
691                                 }
692                                 break loop;
693                             default:
694                                 throw new IllegalStateException(
695                                         "Unknown handshake status: " +
696                                         handshakeStatus);
697                             }
698                         }
699                     }
700                 }
701             }
702         } catch (SSLException e) {
703             success = false;
704             setHandshakeFailure(channel, e);
705             throw e;
706         } finally {
707             bufferPool.release(outNetBuf);
708 
709             if (offered) {
710                 flushPendingEncryptedWrites(context);
711             }
712 
713             if (!success) {
714                 IllegalStateException cause =
715                     new IllegalStateException("SSLEngine already closed");
716                 // Mark all remaining pending writes as failure if anything
717                 // wrong happened before the write requests are wrapped.
718                 // Please note that we do not call setFailure while a lock is
719                 // acquired, to avoid a potential dead lock.
720                 for (;;) {
721                     PendingWrite pendingWrite;
722                     synchronized (pendingUnencryptedWrites) {
723                         pendingWrite = pendingUnencryptedWrites.poll();
724                         if (pendingWrite == null) {
725                             break;
726                         }
727                     }
728 
729                     pendingWrite.future.setFailure(cause);
730                 }
731             }
732         }
733 
734         if (needsUnwrap) {
735             unwrap(context, channel, ChannelBuffers.EMPTY_BUFFER, 0, 0);
736         }
737 
738         if (future == null) {
739             future = succeededFuture(channel);
740         }
741         return future;
742     }

755     private void flushPendingEncryptedWrites(ChannelHandlerContext ctx) {
756         // Avoid possible dead lock and data integrity issue
757         // which is caused by cross communication between more than one channel
758         // in the same VM.
759         if (!pendingEncryptedWritesLock.tryLock()) {
760             return;
761         }
762 
763         try {
764             if (pendingEncryptedWrites.isEmpty()) {
765                 return;
766             }
767 
768             MessageEvent e;
769             while ((e = pendingEncryptedWrites.poll()) != null) {
770                 ctx.sendDownstream(e);
771             }
772         } finally {
773             pendingEncryptedWritesLock.unlock();
774         }
775     }

777     private ChannelFuture wrapNonAppData(ChannelHandlerContext ctx, Channel channel) throws SSLException {
778         ChannelFuture future = null;
779         ByteBuffer outNetBuf = bufferPool.acquire();
780 
781         SSLEngineResult result;
782         try {
783             for (;;) {
784                 synchronized (handshakeLock) {
785                     result = engine.wrap(EMPTY_BUFFER, outNetBuf);
786                 }
787 
788                 if (result.bytesProduced() > 0) {
789                     outNetBuf.flip();
790                     ChannelBuffer msg = ChannelBuffers.buffer(outNetBuf.remaining());
791                     msg.writeBytes(outNetBuf.array(), 0, msg.capacity());
792                     outNetBuf.clear();
793 
794                     future = future(channel);
795                     future.addListener(new ChannelFutureListener() {
796                         public void operationComplete(ChannelFuture future)
797                                 throws Exception {
798                             if (future.getCause() instanceof ClosedChannelException) {
799                                 synchronized (ignoreClosedChannelExceptionLock) {
800                                     ignoreClosedChannelException ++;
801                                 }
802                             }
803                         }
804                     });
805 
806                     write(ctx, future, msg);
807                 }
808 
809                 final HandshakeStatus handshakeStatus = result.getHandshakeStatus();
810                 handleRenegotiation(handshakeStatus);
811                 switch (handshakeStatus) {
812                 case FINISHED:
813                     setHandshakeSuccess(channel);
814                     runDelegatedTasks();
815                     break;
816                 case NEED_TASK:
817                     runDelegatedTasks();
818                     break;
819                 case NEED_UNWRAP:
820                     if (!Thread.holdsLock(handshakeLock)) {
821                         // unwrap shouldn't be called when this method was
822                         // called by unwrap - unwrap will keep running after
823                         // this method returns.
824                         unwrap(ctx, channel, ChannelBuffers.EMPTY_BUFFER, 0, 0);
825                     }
826                     break;
827                 case NOT_HANDSHAKING:
828                 case NEED_WRAP:
829                     break;
830                 default:
831                     throw new IllegalStateException(
832                             "Unexpected handshake status: " + handshakeStatus);
833                 }
834 
835                 if (result.bytesProduced() == 0) {
836                     break;
837                 }
838             }
839         } catch (SSLException e) {
840             setHandshakeFailure(channel, e);
841             throw e;
842         } finally {
843             bufferPool.release(outNetBuf);
844         }
845 
846         if (future == null) {
847             future = succeededFuture(channel);
848         }
849 
850         return future;
851     }

853     private ChannelBuffer unwrap(
854             ChannelHandlerContext ctx, Channel channel, ChannelBuffer buffer, int offset, int length) throws SSLException {
855         ByteBuffer inNetBuf = buffer.toByteBuffer(offset, length);
856         ByteBuffer outAppBuf = bufferPool.acquire();
857 
858         try {
859             boolean needsWrap = false;
860             loop:
861             for (;;) {
862                 SSLEngineResult result;
863                 synchronized (handshakeLock) {
864                     if (!handshaken && !handshaking &&
865                         !engine.getUseClientMode() &&
866                         !engine.isInboundDone() && !engine.isOutboundDone()) {
867                         handshake();
868                     }
869 
870                     try {
871                         result = engine.unwrap(inNetBuf, outAppBuf);
872                     } catch (SSLException e) {
873                         throw e;
874                     }
875 
876                     final HandshakeStatus handshakeStatus = result.getHandshakeStatus();
877                     handleRenegotiation(handshakeStatus);
878                     switch (handshakeStatus) {
879                     case NEED_UNWRAP:
880                         if (inNetBuf.hasRemaining() && !engine.isInboundDone()) {
881                             break;
882                         } else {
883                             break loop;
884                         }
885                     case NEED_WRAP:
886                         wrapNonAppData(ctx, channel);
887                         break;
888                     case NEED_TASK:
889                         runDelegatedTasks();
890                         break;
891                     case FINISHED:
892                         setHandshakeSuccess(channel);
893                         needsWrap = true;
894                         break loop;
895                     case NOT_HANDSHAKING:
896                         needsWrap = true;
897                         break loop;
898                     default:
899                         throw new IllegalStateException(
900                                 "Unknown handshake status: " + handshakeStatus);
901                     }
902                 }
903             }
904 
905             if (needsWrap) {
906                 // wrap() acquires pendingUnencryptedWrites first and then
907                 // handshakeLock.  If handshakeLock is already hold by the
908                 // current thread, calling wrap() will lead to a dead lock
909                 // i.e. pendingUnencryptedWrites -> handshakeLock vs.
910                 //      handshakeLock -> pendingUnencryptedLock -> handshakeLock
911                 //
912                 // There is also a same issue between pendingEncryptedWrites
913                 // and pendingUnencryptedWrites.
914                 if (!Thread.holdsLock(handshakeLock) &&
915                     !pendingEncryptedWritesLock.isHeldByCurrentThread()) {
916                     wrap(ctx, channel);
917                 }
918             }
919 
920             outAppBuf.flip();
921 
922             if (outAppBuf.hasRemaining()) {
923                 ChannelBuffer frame = ChannelBuffers.buffer(outAppBuf.remaining());
924                 frame.writeBytes(outAppBuf.array(), 0, frame.capacity());
925                 return frame;
926             } else {
927                 return null;
928             }
929         } catch (SSLException e) {
930             setHandshakeFailure(channel, e);
931             throw e;
932         } finally {
933             bufferPool.release(outAppBuf);
934         }
935     }

937     private void handleRenegotiation(HandshakeStatus handshakeStatus) {
938         if (handshakeStatus == HandshakeStatus.NOT_HANDSHAKING) {
939             // Not handshaking
940             return;
941         }
942 
943         if (!handshaken) {
944             // Not renegotiation
945             return;
946         }
947 
948         if (handshaking) {
949             // Renegotiation in progress or failed already.
950             // i.e. Renegotiation check has been done already below.
951             return;
952         }
953 
954         if (engine.isInboundDone() || engine.isOutboundDone()) {
955             // Not handshaking but closing.
956             return;
957         }
958 
959         if (isEnableRenegotiation()) {
960             // Continue renegotiation.
961             handshake();
962         } else {
963             // Prevent reentrance of this method.
964             handshaking = true;
965 
966             // Raise an exception.
967             fireExceptionCaught(
968                     ctx, new SSLException(
969                             "renegotiation attempted by peer; " +
970                             "closing the connection"));
971 
972             // Close the connection to stop renegotiation.
973             Channels.close(ctx, succeededFuture(ctx.getChannel()));
974         }
975     }

978         for (;;) {
979             final Runnable task;
980             synchronized (handshakeLock) {
981                 task = engine.getDelegatedTask();

989                 public void run() {
