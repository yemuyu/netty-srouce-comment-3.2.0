37  /**
38   * A {@link Timer} optimized for approximated I/O timeout scheduling.
39   *
40   * <h3>Tick Duration</h3>
41   *
42   * As described with 'approximated', this timer does not execute the scheduled
43   * {@link TimerTask} on time.  {@link HashedWheelTimer}, on every tick, will
44   * check if there are any {@link TimerTask}s behind the schedule and execute
45   * them.
46   * <p>
47   * You can increase or decrease the accuracy of the execution timing by
48   * specifying smaller or larger tick duration in the constructor.  In most
49   * network applications, I/O timeout does not need to be accurate.  Therefore,
50   * the default tick duration is 100 milliseconds and you will not need to try
51   * different configurations in most cases.
52   *
53   * <h3>Ticks per Wheel (Wheel Size)</h3>
54   *
55   * {@link HashedWheelTimer} maintains a data structure called 'wheel'.
56   * To put simply, a wheel is a hash table of {@link TimerTask}s whose hash
57   * function is 'dead line of the task'.  The default number of ticks per wheel
58   * (i.e. the size of the wheel) is 512.  You could specify a larger value
59   * if you are going to schedule a lot of timeouts.
60   *
61   * <h3>Implementation Details</h3>
62   *
63   * {@link HashedWheelTimer} is based on
64   * <a href="http://cseweb.ucsd.edu/users/varghese/">George Varghese</a> and
65   * Tony Lauck's paper,
66   * <a href="http://cseweb.ucsd.edu/users/varghese/PAPERS/twheel.ps.Z">'Hashed
67   * and Hierarchical Timing Wheels: data structures to efficiently implement a
68   * timer facility'</a>.  More comprehensive slides are located
69   * <a href="http://www.cse.wustl.edu/~cdgill/courses/cs6874/TimingWheels.ppt">here</a>.
70   *
71   * @author <a href="http://www.jboss.org/netty/">The Netty Project</a>
72   * @author <a href="http://gleamynode.net/">Trustin Lee</a>
73   * @version $Rev$, $Date$
74   */

155     /**
156      * Creates a new timer.
157      *
158      * @param threadFactory  a {@link ThreadFactory} that creates a
159      *                       background {@link Thread} which is dedicated to
160      *                       {@link TimerTask} execution.
161      * @param tickDuration   the duration between tick
162      * @param unit           the time unit of the {@code tickDuration}
163      * @param ticksPerWheel  the size of the wheel
164      */
165     public HashedWheelTimer(
166             ThreadFactory threadFactory,
167             long tickDuration, TimeUnit unit, int ticksPerWheel) {
168 
169         if (threadFactory == null) {
170             throw new NullPointerException("threadFactory");
171         }
172         if (unit == null) {
173             throw new NullPointerException("unit");
174         }
175         if (tickDuration <= 0) {
176             throw new IllegalArgumentException(
177                     "tickDuration must be greater than 0: " + tickDuration);
178         }
179         if (ticksPerWheel <= 0) {
180             throw new IllegalArgumentException(
181                     "ticksPerWheel must be greater than 0: " + ticksPerWheel);
182         }
183 
184         // Normalize ticksPerWheel to power of two and initialize the wheel.
185         wheel = createWheel(ticksPerWheel);
186         iterators = createIterators(wheel);
187         mask = wheel.length - 1;
188 
189         // Convert tickDuration to milliseconds.
190         this.tickDuration = tickDuration = unit.toMillis(tickDuration);
191 
192         // Prevent overflow.
193         if (tickDuration == Long.MAX_VALUE ||
194                 tickDuration >= Long.MAX_VALUE / wheel.length) {
195             throw new IllegalArgumentException(
196                     "tickDuration is too long: " +
197                     tickDuration +  ' ' + unit);
198         }
199 
200         roundDuration = tickDuration * wheel.length;
201 
202         workerThread = threadFactory.newThread(new ThreadRenamingRunnable(
203                         worker, "Hashed wheel timer #" + id.incrementAndGet()));
204 
205         // Misuse check
206         misuseDetector.increase();
207     }

209     @SuppressWarnings("unchecked")
210     private static Set<HashedWheelTimeout>[] createWheel(int ticksPerWheel) {
211         if (ticksPerWheel <= 0) {
212             throw new IllegalArgumentException(
213                     "ticksPerWheel must be greater than 0: " + ticksPerWheel);
214         }
215         if (ticksPerWheel > 1073741824) {
216             throw new IllegalArgumentException(
217                     "ticksPerWheel may not be greater than 2^30: " + ticksPerWheel);
218         }
219 
220         ticksPerWheel = normalizeTicksPerWheel(ticksPerWheel);
221         Set<HashedWheelTimeout>[] wheel = new Set[ticksPerWheel];
222         for (int i = 0; i < wheel.length; i ++) {
223             wheel[i] = new MapBackedSet<HashedWheelTimeout>(
224                     new ConcurrentIdentityHashMap<HashedWheelTimeout, Boolean>(16, 0.95f, 4));
225         }
226         return wheel;
227     }
228 
229     @SuppressWarnings("unchecked")
230     private static ReusableIterator<HashedWheelTimeout>[] createIterators(Set<HashedWheelTimeout>[] wheel) {
231         ReusableIterator<HashedWheelTimeout>[] iterators = new ReusableIterator[wheel.length];
232         for (int i = 0; i < wheel.length; i ++) {
233             iterators[i] = (ReusableIterator<HashedWheelTimeout>) wheel[i].iterator();
234         }
235         return iterators;
236     }

238     private static int normalizeTicksPerWheel(int ticksPerWheel) {
239         int normalizedTicksPerWheel = 1;
240         while (normalizedTicksPerWheel < ticksPerWheel) {
241             normalizedTicksPerWheel <<= 1;
242         }
243         return normalizedTicksPerWheel;
244     }

246     /**
247      * Starts the background thread explicitly.  The background thread will
248      * start automatically on demand even if you did not call this method.
249      *
250      * @throws IllegalStateException if this timer has been

252      */
253     public synchronized void start() {
254         if (shutdown.get()) {
255             throw new IllegalStateException("cannot be started once stopped");
256         }
257 
258         if (!workerThread.isAlive()) {
259             workerThread.start();
260         }
261     }

263     public synchronized Set<Timeout> stop() {
264         if (!shutdown.compareAndSet(false, true)) {
265             return Collections.emptySet();
266         }
267 
268         boolean interrupted = false;
269         while (workerThread.isAlive()) {
270             workerThread.interrupt();
271             try {
272                 workerThread.join(100);
273             } catch (InterruptedException e) {
274                 interrupted = true;
275             }
276         }
277 
278         if (interrupted) {
279             Thread.currentThread().interrupt();
280         }
281 
282         misuseDetector.decrease();
283 
284         Set<Timeout> unprocessedTimeouts = new HashSet<Timeout>();
285         for (Set<HashedWheelTimeout> bucket: wheel) {
286             unprocessedTimeouts.addAll(bucket);
287             bucket.clear();
288         }
289 
290         return Collections.unmodifiableSet(unprocessedTimeouts);
291     }

293     public Timeout newTimeout(TimerTask task, long delay, TimeUnit unit) {
294         final long currentTime = System.currentTimeMillis();
295 
296         if (task == null) {
297             throw new NullPointerException("task");
298         }
299         if (unit == null) {
300             throw new NullPointerException("unit");
301         }
302 
303         delay = unit.toMillis(delay);
304         if (delay < tickDuration) {
305             delay = tickDuration;
306         }
307 
308         if (!workerThread.isAlive()) {
309             start();
310         }
311 
312         // Prepare the required parameters to create the timeout object.
313         HashedWheelTimeout timeout;
314         final long lastRoundDelay = delay % roundDuration;
315         final long lastTickDelay = delay % tickDuration;
316         final long relativeIndex =
317             lastRoundDelay / tickDuration + (lastTickDelay != 0? 1 : 0);
318         final long deadline = currentTime + delay;
319 
320         final long remainingRounds =
321             delay / roundDuration - (delay % roundDuration == 0? 1 : 0);
322 
323         // Add the timeout to the wheel.
324         lock.readLock().lock();
325         try {
326             timeout =
327                 new HashedWheelTimeout(
328                         task, deadline,
329                         (int) (wheelCursor + relativeIndex & mask),
330                         remainingRounds);
331 
332             wheel[timeout.stopIndex].add(timeout);
333         } finally {
334             lock.readLock().unlock();
335         }
336 
337         return timeout;
338     }

340     private final class Worker implements Runnable {
341 
342         private long startTime;
343         private long tick;
344 
345         Worker() {
346             super();
347         }
348 
349         public void run() {
350             List<HashedWheelTimeout> expiredTimeouts =
351                 new ArrayList<HashedWheelTimeout>();
352 
353             startTime = System.currentTimeMillis();
354             tick = 1;
355 
356             while (!shutdown.get()) {
357                 waitForNextTick();
358                 fetchExpiredTimeouts(expiredTimeouts);
359                 notifyExpiredTimeouts(expiredTimeouts);
360             }
361         }
362 
363         private void fetchExpiredTimeouts(
364                 List<HashedWheelTimeout> expiredTimeouts) {
365 
366             // Find the expired timeouts and decrease the round counter
367             // if necessary.  Note that we don't send the notification
368             // immediately to make sure the listeners are called without
369             // an exclusive lock.
370             lock.writeLock().lock();
371             try {
372                 int oldBucketHead = wheelCursor;
373                 int newBucketHead = oldBucketHead + 1 & mask;
374                 wheelCursor = newBucketHead;
375 
376                 ReusableIterator<HashedWheelTimeout> i = iterators[oldBucketHead];
377                 fetchExpiredTimeouts(expiredTimeouts, i);
378             } finally {
379                 lock.writeLock().unlock();
380             }
381         }
382 
383         private void fetchExpiredTimeouts(
384                 List<HashedWheelTimeout> expiredTimeouts,
385                 ReusableIterator<HashedWheelTimeout> i) {
386 
387             long currentDeadline = System.currentTimeMillis() + tickDuration;
388             i.rewind();
389             while (i.hasNext()) {
390                 HashedWheelTimeout timeout = i.next();
391                 if (timeout.remainingRounds <= 0) {
392                     if (timeout.deadline < currentDeadline) {
393                         i.remove();
394                         expiredTimeouts.add(timeout);
395                     } else {
396                         // A rare case where a timeout is put for the next
397                         // round: just wait for the next round.
398                     }
399                 } else {
400                     timeout.remainingRounds --;
401                 }
402             }
403         }
404 
405         private void notifyExpiredTimeouts(
406                 List<HashedWheelTimeout> expiredTimeouts) {
407             // Notify the expired timeouts.
408             for (int i = expiredTimeouts.size() - 1; i >= 0; i --) {
409                 expiredTimeouts.get(i).expire();
410             }
411 
412             // Clean up the temporary list.
413             expiredTimeouts.clear();
414         }
415 
416         private void waitForNextTick() {
417             for (;;) {
418                 final long currentTime = System.currentTimeMillis();
419                 final long sleepTime = tickDuration * tick - (currentTime - startTime);
420 
421                 if (sleepTime <= 0) {
422                     break;
423                 }
424 
425                 try {
426                     Thread.sleep(sleepTime);
427                 } catch (InterruptedException e) {
428                     if (shutdown.get()) {
429                         return;
430                     }
431                 }
432             }
433 
434             // Reset the tick if overflow is expected.
435             if (tickDuration * tick > Long.MAX_VALUE - tickDuration) {
436                 startTime = System.currentTimeMillis();
437                 tick = 1;
438             } else {
439                 // Increase the tick if overflow is not likely to happen.
440                 tick ++;
441             }
442         }
443     }

445     private final class HashedWheelTimeout implements Timeout {
446 
447         private final TimerTask task;
448         final int stopIndex;
449         final long deadline;
450         volatile long remainingRounds;
451         private volatile boolean cancelled;
452 
453         HashedWheelTimeout(
454                 TimerTask task, long deadline, int stopIndex, long remainingRounds) {
455             this.task = task;
456             this.deadline = deadline;
457             this.stopIndex = stopIndex;
458             this.remainingRounds = remainingRounds;
459         }
460 
461         public Timer getTimer() {
462             return HashedWheelTimer.this;
463         }
464 
465         public TimerTask getTask() {
466             return task;
467         }
468 
469         public void cancel() {
470             if (isExpired()) {
471                 return;
472             }
473 
474             cancelled = true;
475 
476             // Might be called more than once, but doesn't matter.
477             wheel[stopIndex].remove(this);
478         }
479 
480         public boolean isCancelled() {
481             return cancelled;
482         }
483 
484         public boolean isExpired() {
485             return cancelled || System.currentTimeMillis() > deadline;
486         }
487 
488         public void expire() {
489             if (cancelled) {
490                 return;
491             }
492 
493             try {
494                 task.run(this);
495             } catch (Throwable t) {
496                 logger.warn(
497                         "An exception was thrown by " +
498                         TimerTask.class.getSimpleName() + ".", t);
499             }
500         }
501 
502         @Override
503         public String toString() {
504             long currentTime = System.currentTimeMillis();
505             long remaining = deadline - currentTime;
506 
507             StringBuilder buf = new StringBuilder(192);
508             buf.append(getClass().getSimpleName());
509             buf.append('(');
510 
511             buf.append("deadline: ");
512             if (remaining > 0) {
513                 buf.append(remaining);
514                 buf.append(" ms later, ");
515             } else if (remaining < 0) {
516                 buf.append(-remaining);
517                 buf.append(" ms ago, ");
518             } else {
519                 buf.append("now, ");
520             }
521 
522             if (isCancelled()) {
523                 buf.append (", cancelled");
524             }
525 
526             return buf.append(')').toString();
527         }
528     }
